### Task 2: `AsPool` trait + blanket `impl GenericClient for &P`

**Files:**
- Modify: `crates/sentinel-driver/src/generic_client.rs`

- [ ] **Step 1: Define `AsPool`**

After the existing `GenericClient` trait + impls in `generic_client.rs`, add:

```rust
/// Types from which a `Connection` can be borrowed for the duration of one
/// closure. The blanket `impl GenericClient for &P` builds on this.
///
/// Implemented on `Pool` and re-implementable by user wrappers (e.g. a
/// load-balanced multi-pool router).
pub trait AsPool {
    /// Acquire a connection, run `f` on it, release on drop.
    fn with_conn<'a, R, F>(
        &'a self,
        f: F,
    ) -> impl std::future::Future<Output = Result<R>> + Send + 'a
    where
        F: FnOnce(&'a mut Connection) -> futures::future::BoxFuture<'a, Result<R>> + Send + 'a,
        R: Send + 'a;
}
```

(Implementation note: using a `BoxFuture` here side-steps the
`for<'a> AsyncFnOnce` trait bound, which doesn't yet have stable
object-safety in stable Rust. `BoxFuture` is `Pin<Box<dyn Future>>` —
one heap allocation per macro call, negligible vs the network round-trip.)

If the workspace doesn't already use `futures::future::BoxFuture`, add
`futures = "0.3"` to `[dependencies]` in
`crates/sentinel-driver/Cargo.toml`. Verify with
`grep -rn "BoxFuture\|^futures" crates/sentinel-driver/`.

impl crate::AsPool for Pool {
    fn with_conn<'a, R, F>(
        &'a self,
        f: F,
    ) -> impl std::future::Future<Output = crate::Result<R>> + Send + 'a
    where
        F: FnOnce(&'a mut crate::Connection) -> futures::future::BoxFuture<'a, crate::Result<R>>
            + Send
            + 'a,
        R: Send + 'a,
    {
        async move {
            let mut pooled = self.acquire().await?;
            // PooledConnection derefs to Connection; we need a &mut Connection.
            let conn: &mut crate::Connection = &mut *pooled;
            f(conn).await
        }
    }
}

    ///
    /// Unlike `fetch_all`, this does not materialise rows in memory.
    /// The stream borrows the connection until dropped — the caller
    /// cannot reuse the connection for other queries until then.
    /// PG portals are tx-scoped; the stream opens an implicit

        Ok(conn.query_stream(self.handle.sql, &plain).await?)
    }

```rust
//! Template-database lifecycle. Builds `_sntl_tmpl_<hash>` once per
//! process per migrations directory; clones it per test.

use std::collections::HashMap;
use std::path::Path;
use std::sync::OnceLock;

use sentinel_driver::pool::config::PoolConfig;
use sentinel_driver::{Config, Connection, Pool};
use sha2::{Digest, Sha256};
use tokio::sync::Mutex;

#[derive(Debug)]
pub struct TemplateDb {
    pub db_name: String,
    pub admin_url: String,
}

/// Build (once) or look up the template DB for the given migrations
/// directory. Returns the template DB name.
pub async fn build_or_get(
    admin_url: &str,
    migrations_dir: Option<&Path>,
) -> anyhow::Result<TemplateDb> {
    static CACHE: OnceLock<Mutex<HashMap<String, TemplateDb>>> = OnceLock::new();
    let cache = CACHE.get_or_init(|| Mutex::new(HashMap::new()));

    let key = template_key(migrations_dir)?;
    let db_name = format!("_sntl_tmpl_{key}");

    let mut guard = cache.lock().await;
    if let Some(existing) = guard.get(&key) {
        return Ok(TemplateDb {
            db_name: existing.db_name.clone(),
            admin_url: existing.admin_url.clone(),
        });
    }

    // Build it.
    create_template(admin_url, &db_name, migrations_dir).await?;
    let t = TemplateDb {
        db_name: db_name.clone(),
        admin_url: admin_url.to_string(),
    };
    guard.insert(key, TemplateDb {
        db_name: db_name.clone(),
        admin_url: admin_url.to_string(),
    });
    Ok(t)
}

fn template_key(migrations_dir: Option<&Path>) -> anyhow::Result<String> {
    let canon = match migrations_dir {
        Some(p) => p
            .canonicalize()
            .map(|c| c.to_string_lossy().into_owned())
            .unwrap_or_else(|_| String::from("none")),
        None => String::from("none"),
    };
    let mut h = Sha256::new();
    h.update(canon.as_bytes());
    let digest = h.finalize();
    Ok(hex::encode(&digest[..4]))
}

async fn create_template(
    admin_url: &str,
    db_name: &str,
    migrations_dir: Option<&Path>,
) -> anyhow::Result<()> {
    // Connect to admin DB (postgres) and create the template DB.
    let cfg = Config::parse(admin_url)?;
    let mut admin = Connection::connect(cfg).await?;
    admin
        .execute("SET client_min_messages = ERROR", &[])
        .await?;
    admin
        .execute(&format!("DROP DATABASE IF EXISTS {db_name}"), &[])
        .await?;
    admin
        .execute(&format!("CREATE DATABASE {db_name}"), &[])
        .await?;
    drop(admin);

    // Connect to the new DB and apply migrations.
    if let Some(dir) = migrations_dir {
        let mut tmpl_url = url::Url::parse(admin_url)?;
        tmpl_url.set_path(&format!("/{db_name}"));
        let tmpl_cfg = Config::parse(tmpl_url.as_str())?;
        let pool = Pool::new(tmpl_cfg, PoolConfig::new().max_connections(2));
        let migrator = sntl_migrate::Migrator::from_dir(dir)?;
        migrator.run(&pool).await?;
    }

    // Mark it as a template so CREATE DATABASE ... TEMPLATE works.
    let cfg = Config::parse(admin_url)?;
    let mut admin = Connection::connect(cfg).await?;
    admin
        .execute(
            &format!("UPDATE pg_database SET datistemplate = true WHERE datname = '{db_name}'"),
            &[],
        )
        .await?;
    Ok(())
}

/// Create a fresh DB from the template. Returns the new DB name.
pub async fn clone_into(template: &TemplateDb, suffix: &str) -> anyhow::Result<String> {
    let new_db = format!("_sntl_{suffix}");
    let cfg = Config::parse(&template.admin_url)?;
    let mut admin = Connection::connect(cfg).await?;
    admin
        .execute("SET client_min_messages = ERROR", &[])
        .await?;
    admin
        .execute(
            &format!("CREATE DATABASE {} TEMPLATE {}", new_db, template.db_name),
            &[],
        )
        .await?;
    Ok(new_db)
}

/// Best-effort drop. Logs but does not panic on failure.
pub async fn drop_db(admin_url: &str, db_name: &str) {
    let cfg = match Config::parse(admin_url) {
        Ok(c) => c,
        Err(_) => return,
    };
    if let Ok(mut admin) = Connection::connect(cfg).await {
        let _ = admin.execute("SET client_min_messages = ERROR", &[]).await;
        let _ = admin
            .execute(&format!("DROP DATABASE IF EXISTS {db_name}"), &[])
            .await;
    }
}

//! Apply fixture SQL files in declared order, inside one transaction.

use std::path::PathBuf;

use sentinel_driver::Connection;

/// Apply each named fixture (e.g. "users" → `tests/fixtures/users.sql`)
/// inside a single transaction. Rolls back on error.
pub async fn apply(
    conn: &mut Connection,
    fixtures_dir: &std::path::Path,
    names: &[&str],
) -> anyhow::Result<()> {
    if names.is_empty() {
        return Ok(());
    }

    conn.execute("BEGIN", &[]).await?;
    for name in names {
        let mut p: PathBuf = fixtures_dir.to_path_buf();
        p.push(format!("{name}.sql"));
        let sql = std::fs::read_to_string(&p).map_err(|e| {
            anyhow::anyhow!("read fixture {}: {}", p.display(), e)
        })?;
        if let Err(e) = conn.execute(&sql, &[]).await {
            conn.execute("ROLLBACK", &[]).await.ok();
            return Err(e.into());
        }
    }
    conn.execute("COMMIT", &[]).await?;
    Ok(())
}
```

    let test_db = template::clone_into(&template, &suffix).await

    if !cfg.fixtures.is_empty() {

        fixtures::apply(&mut *conn, &fixtures_root, cfg.fixtures)
            .await
            .unwrap_or_else(|e| panic!("sntl::test: apply fixtures failed: {e}"));
    }

    // Run body. Catch panics so we can leak the DB on failure.
    let result = body(pool).await;

    let keep = std::env::var("SNTL_TEST_KEEP_DBS").is_ok();
    if let Err(e) = result {

    }

use proc_macro2::TokenStream;
use proc_macro_error2::abort;
use quote::quote;
use syn::parse::{Parse, ParseStream};
use syn::punctuated::Punctuated;
use syn::{ItemFn, LitStr, Meta, Token};

/// Parses `migrations = "./migrations", fixtures("users", "posts")`.
struct Args {
    migrations_dir: Option<String>,
    fixtures: Vec<String>,
}

impl Parse for Args {
    fn parse(input: ParseStream) -> syn::Result<Self> {
        let mut migrations_dir = None;
        let mut fixtures = Vec::new();

        if input.is_empty() {
            return Ok(Self { migrations_dir, fixtures });
        }

        let metas: Punctuated<Meta, Token![,]> = Punctuated::parse_terminated(input)?;
        for m in metas {
            match &m {
                Meta::NameValue(nv) if nv.path.is_ident("migrations") => {
                    if let syn::Expr::Lit(syn::ExprLit { lit: syn::Lit::Str(s), .. }) = &nv.value {
                        migrations_dir = Some(s.value());
                    } else {
                        return Err(syn::Error::new_spanned(&nv.value, "expected string literal"));
                    }
                }
                Meta::List(l) if l.path.is_ident("fixtures") => {
                    let names: Punctuated<LitStr, Token![,]> =
                        l.parse_args_with(Punctuated::parse_terminated)?;
                    fixtures = names.into_iter().map(|s| s.value()).collect();
                }
                _ => {
                    return Err(syn::Error::new_spanned(&m, "unknown sntl::test argument"));
                }
            }
        }

        Ok(Self { migrations_dir, fixtures })
    }
}

pub fn expand(attr: TokenStream, item: TokenStream) -> TokenStream {
    let args: Args = match syn::parse2(attr) {
        Ok(a) => a,
        Err(e) => abort!(e.span(), "{}", e),
    };
    let mut input_fn: ItemFn = match syn::parse2(item) {
        Ok(f) => f,
        Err(e) => abort!(e.span(), "{}", e),
    };

    let fn_name = input_fn.sig.ident.clone();
    let fn_name_str = fn_name.to_string();
    let asyncness = input_fn.sig.asyncness;
    if asyncness.is_none() {
        abort!(input_fn.sig.fn_token.span, "#[sntl::test] requires async fn");
    }
    let body = input_fn.block.clone();
    let inputs = input_fn.sig.inputs.clone();

    let migrations_dir_lit = match args.migrations_dir {
        Some(s) => quote! { Some(#s) },
        None => quote! { None },
    };
    let fixtures_lits = args.fixtures.iter().map(|s| quote! { #s });

    // Strip async + body — we'll rewrite into a sync test entry that
    // builds its own runtime and calls sntl::testing::run.
    input_fn.sig.asyncness = None;
    input_fn.block = syn::parse2(quote! {
        {
            let cfg = ::sntl::testing::run::TestConfig {
                test_name: #fn_name_str,
                migrations_dir: #migrations_dir_lit,
                fixtures_dir: None,
                fixtures: &[ #(#fixtures_lits),* ],
